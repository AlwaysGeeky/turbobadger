258 class SizeConstraints
259 {
260 public:
261  static const int NO_RESTRICTION = 10000;
262 
264  int available_w, available_h;
265 
267  SizeConstraints(int w, int h) : available_w(w), available_h(h) {}
268 
270  SizeConstraints() : available_w(NO_RESTRICTION), available_h(NO_RESTRICTION) {}
271 
273  SizeConstraints ConstrainByPadding(int horizontal_padding, int vertical_padding) const
274  {
275  return SizeConstraints(available_w == NO_RESTRICTION ? NO_RESTRICTION : available_w - horizontal_padding,
276  available_h == NO_RESTRICTION ? NO_RESTRICTION : available_h - vertical_padding);
277  }
278 
280  SizeConstraints ConstrainByLayoutParams(const LayoutParams &lp) const
281  {
282  return SizeConstraints(ConstrainByLPMax(available_w, lp.min_w, lp.max_w),
283  ConstrainByLPMax(available_h, lp.min_h, lp.max_h));
284  }
285 
286  bool operator == (const SizeConstraints &sc) const { return available_w == sc.available_w &&
287  available_h == sc.available_h; }
288 
289 private:
290  int ConstrainByLPMax(int constraint, int lp_min, int lp_max) const
291  {
292  if (constraint == NO_RESTRICTION)
293  return lp_max != LayoutParams::UNSPECIFIED ? lp_max : NO_RESTRICTION;
294  int ret = constraint;
295  if (lp_min != LayoutParams::UNSPECIFIED)
296  ret = MAX(ret, lp_min);
297  if (lp_max != LayoutParams::UNSPECIFIED)
298  ret = MIN(ret, lp_max);
299  return ret;
300  }
301 };
